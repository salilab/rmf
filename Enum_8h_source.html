42 /** A general purpose Enum that has associated names and is type checked in
43  * python.*/
44 template <class TagT>
45 class Enum {
46  int i_;
47  int compare(const Enum<TagT>& o) const {
48  if (i_ < o.i_)
49  return -1;
50  else if (i_ > o.i_)
51  return 1;
52  else
53  return 0;
54  }
55  std::string get_string() const { return TagT::get_to().find(i_)->second; }
56 
57  public:
58  typedef TagT Tag;
59  Enum() : i_(-1) {}
60  explicit Enum(int i) : i_(i) {
61  RMF_USAGE_CHECK(TagT::get_to().find(i) != TagT::get_to().end(),
62  "Enum value not defined");
63  }
64  Enum(std::string name) {
65  RMF_USAGE_CHECK(TagT::get_from().find(name) != TagT::get_from().end(),
66  "Enum name not defined");
67  i_ = TagT::get_from().find(name)->second;
68  }
69  Enum(int i, std::string name) : i_(i) {
70  TagT::get_to()[i] = name;
71  TagT::get_from()[name] = i;
72  }
73  RMF_COMPARISONS(Enum);
74  RMF_HASHABLE(Enum, return i_);
75 #if !defined(RMF_DOXYGEN) && !defined(SWIG)
76  void show(std::ostream& out) const { out << get_string(); }
77  operator int() const { return i_; }
78 #endif
79 };
