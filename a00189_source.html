34 /** Wrap an HDF5 data set. Typedefs and python types are provided for
35  data sets in 1,2, and 3 dimensions with all the supported types. They are
36  named as
37  RMF::HDF5IndexDataSet2D (or RMF.HDF5IndexDataSet2).
38  See
39  \external{http://www.hdfgroup.org/HDF5/doc/UG/UG_frame10Datasets.html,
40  the HDF5 manual} for more information.

54  int compare(const ConstDataSetD<TypeTraits, D>& o) const {
55  // not great, but...
56  if (data_ && !o.data_)
57  return -1;
58  else if (o.data_ && !data_)
59  return 1;
60  else if (!o.data_ && !data_)
61  return 0;
62  else if (get_name() < o.get_name())
63  return -1;
64  else if (get_name() > o.get_name())
65  return 1;
66  else
67  return 0;
68  }
