107 std::pair<double, std::size_t> create_frame(RMF::FileHandle fh,
108  RMF::decorator::ParticleFactory ipf,
109  const RMF::NodeIDs& atoms,
110  int frame) {
111  RMF::Vector3 ret(0, 0, 0);
112  std::size_t total_size = 0;
113  RMF_FOREACH(RMF::NodeID n, atoms) {
114  RMF::Vector3 v((n.get_index() + 0 + frame) / 17.0,
115  (n.get_index() + 1 + frame) / 19.0,
116  (n.get_index() + 2 + frame) / 23.0);
117  ret[0] += v[0];
118  ret[1] += v[1];
119  ret[2] += v[2];
120  ipf.get(fh.get_node(n)).set_frame_coordinates(v);
121  total_size += sizeof(float) * 3;
122  }
123  return std::make_pair(ret[0] + ret[1] + ret[2], total_size);
124 }

126 boost::tuple<double, std::size_t, std::size_t> create(RMF::FileHandle file) {
127  std::size_t hierarchy_size = create_hierarchy(file);
128  RMF::NodeIDs atoms;
129  RMF_FOREACH(RMF::NodeID n, file.get_node_ids()) {
130  if (file.get_node(n).get_children().empty()) {
131  atoms.push_back(n);
132  }
133  }
134  RMF::decorator::ParticleFactory ipf(file);
135  double check_value = 0;
136  std::size_t frame_size = 0;
137  for (unsigned int i = 0; i < 20; ++i) {
138  file.add_frame("frame", RMF::FRAME);
139  std::pair<double, std::size_t> cur = create_frame(file, ipf, atoms, i);
140  check_value += cur.first;
141  frame_size += cur.second;
142  }
143  return boost::make_tuple(check_value, hierarchy_size, frame_size);
144 }

146 double traverse(RMF::FileConstHandle file) {
147  double ret = 0;
148  RMF::NodeConstHandles queue(1, file.get_root_node());
149  RMF::decorator::ParticleFactory ipcf(file);
150  do {
151  RMF::NodeConstHandle cur = queue.back();
152  queue.pop_back();
153  if (ipcf.get_is(cur)) {
154  ret += ipcf.get(cur).get_radius();
155  }
156  RMF::NodeConstHandles children = cur.get_children();
157  queue.insert(queue.end(), children.begin(), children.end());
158  } while (!queue.empty());
159  return ret;
160 }

162 double load(RMF::FileConstHandle file, const RMF::NodeIDs& nodes) {
163  RMF::decorator::IntermediateParticleFactory ipcf(file);
164  RMF::Vector3 v(0, 0, 0);
165  RMF_FOREACH(RMF::FrameID fr, file.get_frames()) {
166  file.set_current_frame(fr);
167  RMF_FOREACH(RMF::NodeID n, nodes) {
168  RMF::Vector3 cur = ipcf.get(file.get_node(n)).get_coordinates();
169  v[0] += cur[0];
170  v[1] += cur[1];
171  v[2] += cur[2];
172  }
173  }
174  return v[0] + v[1] + v[2];
175 }
